Source Code Firmware NOVA

main.ino - Program Utama 
#include <WiFi.h>
#include <Adafruit_SSD1306.h>
#include <Wire.h>
#include <RTClib.h>

// Pin definitions
#define MIC_PIN 34
#define BTN_LEFT 13
#define BTN_RIGHT 14
#define BTN_SELECT 27
#define LED_PIN 2

// OLED definitions
#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_RESET -1
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

// RTC object
RTC_DS3231 rtc;

// Menu states
enum MenuState {
    MENU_MAIN,
    MENU_CLOCK,
    MENU_SOUND,
    MENU_GAME,
    MENU_SETTINGS
};

MenuState currentState = MENU_MAIN;
int menuIndex = 0;
const char* mainMenu[] = {"Jam", "Suara", "Game", "Setting"};

void setup() {
    Serial.begin(115200);
    
    // Initialize pins
    pinMode(MIC_PIN, INPUT);
    pinMode(BTN_LEFT, INPUT_PULLUP);
    pinMode(BTN_RIGHT, INPUT_PULLUP);
    pinMode(BTN_SELECT, INPUT_PULLUP);
    pinMode(LED_PIN, OUTPUT);
    
    // Initialize OLED
    if(!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
        Serial.println("OLED tidak ditemukan!");
        while(1);
    }
    
    // Initialize RTC
    if (!rtc.begin()) {
        Serial.println("RTC tidak ditemukan!");
        while(1);
    }
    
    if (rtc.lostPower()) {
        rtc.adjust(DateTime(F(__DATE__), F(__TIME__)));
    }
    
    display.clearDisplay();
    display.setTextSize(1);
    display.setTextColor(SSD1306_WHITE);
    
    // Boot animation
    bootAnimation();
}

void loop() {
    handleButtons();
    
    switch(currentState) {
        case MENU_MAIN:
            drawMainMenu();
            break;
        case MENU_CLOCK:
            drawClock();
            break;
        case MENU_SOUND:
            drawSoundMeter();
            break;
        case MENU_GAME:
            drawGame();
            break;
        case MENU_SETTINGS:
            drawSettings();
            break;
    }
    
    delay(50);
}

void handleButtons() {
    static unsigned long lastDebounceTime = 0;
    const unsigned long debounceDelay = 200;
    
    if (digitalRead(BTN_LEFT) == LOW) {
        if (millis() - lastDebounceTime > debounceDelay) {
            lastDebounceTime = millis();
            if (currentState == MENU_MAIN) {
                menuIndex = (menuIndex - 1 + 4) % 4;
            }
        }
    }
    
    if (digitalRead(BTN_RIGHT) == LOW) {
        if (millis() - lastDebounceTime > debounceDelay) {
            lastDebounceTime = millis();
            if (currentState == MENU_MAIN) {
                menuIndex = (menuIndex + 1) % 4;
            }
        }
    }
    
    if (digitalRead(BTN_SELECT) == LOW) {
        if (millis() - lastDebounceTime > debounceDelay) {
            lastDebounceTime = millis();
            if (currentState == MENU_MAIN) {
                // Enter selected menu
                switch(menuIndex) {
                    case 0: currentState = MENU_CLOCK; break;
                    case 1: currentState = MENU_SOUND; break;
                    case 2: currentState = MENU_GAME; break;
                    case 3: currentState = MENU_SETTINGS; break;
                }
            } else {
                // Back to main menu
                currentState = MENU_MAIN;
            }
        }
    }
}

void drawMainMenu() {
    display.clearDisplay();
    display.setTextSize(2);
    display.setCursor(20, 10);
    display.println("NOVA");
    
    display.setTextSize(1);
    display.setCursor(0, 40);
    
    for(int i = 0; i < 4; i++) {
        if(i == menuIndex) {
            display.print(">");
        } else {
            display.print(" ");
        }
        display.print(mainMenu[i]);
        display.print("  ");
    }
    
    display.display();
}

void drawClock() {
    DateTime now = rtc.now();
    
    display.clearDisplay();
    display.setTextSize(2);
    display.setCursor(20, 10);
    
    // Format: HH:MM:SS
    char timeStr[9];
    sprintf(timeStr, "%02d:%02d:%02d", now.hour(), now.minute(), now.second());
    display.println(timeStr);
    
    display.setTextSize(1);
    display.setCursor(25, 40);
    
    // Format: DD/MM/YYYY
    char dateStr[11];
    sprintf(dateStr, "%02d/%02d/%04d", now.day(), now.month(), now.year());
    display.println(dateStr);
    
    display.display();
}

void drawSoundMeter() {
    int soundValue = analogRead(MIC_PIN);
    int barLength = map(soundValue, 0, 4095, 0, 120);
    
    display.clearDisplay();
    display.setTextSize(1);
    display.setCursor(0, 0);
    display.print("Level Suara:");
    
    display.setCursor(0, 20);
    display.print("Nilai: ");
    display.print(soundValue);
    
    // Draw bar graph
    display.fillRect(0, 40, barLength, 15, SSD1306_WHITE);
    
    // Draw emoji based on sound level
    display.setCursor(0, 55);
    if(soundValue < 1000) {
        display.println("😊 Tenang");
    } else if(soundValue < 2000) {
        display.println("😐 Normal");
    } else if(soundValue < 3000) {
        display.println("😲 Ramai");
    } else {
        display.println("😱 BERISIK!");
    }
    
    display.display();
}

void drawGame() {
    static int score = 0;
    static unsigned long lastGameUpdate = 0;
    
    if(millis() - lastGameUpdate > 100) {
        lastGameUpdate = millis();
        
        // Simple reaction game
        if(digitalRead(BTN_SELECT) == LOW) {
            score++;
            digitalWrite(LED_PIN, HIGH);
            delay(50);
            digitalWrite(LED_PIN, LOW);
        }
    }
    
    display.clearDisplay();
    display.setTextSize(1);
    display.setCursor(0, 10);
    display.println("Reaction Game");
    
    display.setCursor(0, 30);
    display.print("Score: ");
    display.print(score);
    
    display.setCursor(0, 45);
    display.println("Tekan SELECT!");
    
    display.display();
}

void drawSettings() {
    display.clearDisplay();
    display.setTextSize(1);
    display.setCursor(0, 10);
    display.println("Settings:");
    display.setCursor(0, 25);
    display.println("- Brightness");
    display.setCursor(0, 35);
    display.println- Sound Level");
    display.setCursor(0, 45);
    display.println("- About");
    display.display();
}

void bootAnimation() {
    for(int i = 0; i < 3; i++) {
        display.clearDisplay();
        display.setTextSize(2);
        display.setCursor(20, 20);
        display.println("NOVA");
        display.display();
        delay(500);
        
        display.clearDisplay();
        display.setTextSize(1);
        display.setCursor(10, 20);
        display.println("Booting...");
        display.display();
        delay(500);
    }
}

Cara Install Library

// Melalui Arduino Library Manager
1. Buka Arduino IDE
2. Tools → Manage Libraries
3. Cari "Adafruit SSD1306"
4. Klik Install
5. Ulangi untuk library lainnya

// Atau melalui GitHub
1. Download ZIP dari repository
2. Sketch → Include Library → Add .ZIP Library
3. Pilih file ZIP yang sudah didownload
