// Fungsi double click didefinisikan hanya di sini
bool isButtonDoubleClicked(uint8_t pin) {
  static unsigned long lastPressTime[10] = {0};
  static bool firstPressDetected[10] = {false};
  static unsigned long debounceTime[10] = {0};
  
  // Baca state button (LOW karena INPUT_PULLUP)
  bool buttonState = !digitalRead(pin);
  unsigned long currentTime = millis();
  
  // Debounce sederhana
  if (buttonState && (currentTime - debounceTime[pin] > 50)) {
    debounceTime[pin] = currentTime;
    
    if (!firstPressDetected[pin]) {
      // First press
      firstPressDetected[pin] = true;
      lastPressTime[pin] = currentTime;
      Serial.println("First click detected");
    } else {
      // Check for second press within DOUBLE_CLICK_TIME
      if (currentTime - lastPressTime[pin] <= DOUBLE_CLICK_TIME) {
        firstPressDetected[pin] = false;
        Serial.println("Double click detected!");
        return true;
      }
    }
  }
  
  // Reset if too much time has passed since first press (300ms)
  if (firstPressDetected[pin] && (currentTime - lastPressTime[pin] > DOUBLE_CLICK_TIME)) {
    firstPressDetected[pin] = false;
    Serial.println("First click timeout");
  }
  
  return false;
}

void setup() {

  pinMode(BTN_SELECT, INPUT_PULLUP);
  pinMode(BTN_BACK, INPUT_PULLUP);

  // Initialize RGB LED
  Serial.println("Initializing RGB LED...");
  pixels.begin();
  pixels.clear();
  pixels.show();
  
  // Boot animation RGB
  bootAnimationRGB();

  // Initialize OLED dengan konfigurasi SPI 7-pin
  Serial.println("Initializing OLED...");
  if (!display.begin(SSD1306_SWITCHCAPVCC)) {  // Perbaiki: SSD1306_SWITCHCAPVCC
    Serial.println("OLED initialization failed!");
    while (1);
  }
  Serial.println("OLED initialized successfully!");
  
  display.clearDisplay();
  display.setTextColor(SSD1306_WHITE);
  display.setTextSize(1);
  display.setCursor(0, 0);
  display.println("System Starting...");
  display.display();
  delay(1000);

  // Initialize RTC
  Serial.println("Initializing RTC...");
  Wire.begin(21, 20); // SDA=21, SCL=20 untuk ESP32-S3 (perbaiki komentar)
  if (!rtc.begin()) {
    Serial.println("Couldn't find RTC!");
    display.clearDisplay();
    display.setTextSize(2);
    display.setCursor(0, 0);
    display.println("RTC ERROR!");
    display.setTextSize(1);
    display.setCursor(0, 20);
    display.println("Check your RTC!");
    display.display();
  } else {
    if (rtc.lostPower()) {
      Serial.println("RTC lost power, setting time!");
      rtc.adjust(DateTime(F(__DATE__), F(__TIME__)));

    Serial.println("RTC initialized!");
  }

  Serial.println("Initializing I2S...");
  setupI2S();

  // Play boot sequence dengan suara Zelda

void updateRGB() {
  unsigned long currentTime = millis();
  
  switch (currentRGBMode) {
    case RGB_MODE_CHOSEN:
      // Tetap pada warna yang dipilih
      setRGBColor(selectedColor);
      break;
      
    case RGB_MODE_RANDOM:
      // Ganti warna random setiap interval yang ditentukan
      if (currentTime - lastColorChange > (colorChangeInterval * 1000)) {
        selectedColor = random(0, 7);
        setRGBColor(selectedColor);
        lastColorChange = currentTime;
        Serial.print("RGB Random Color: ");
        Serial.println(colorList[selectedColor]);
      }
      break;
      
    case RGB_MODE_LOOP:
      // Ganti warna berurutan setiap interval yang ditentukan
      if (currentTime - lastColorChange > (colorChangeInterval * 1000)) {
        currentLoopColor = (currentLoopColor + 1) % 7;
        selectedColor = currentLoopColor;
        setRGBColor(selectedColor);
        lastColorChange = currentTime;
        Serial.print("RGB Loop Color: ");
        Serial.println(colorList[selectedColor]);
      }
      break;
  }
}

  isDucking = false;
  obstacleX = 128;
  score = 0;
  lastScoreTime = millis();
  obstaclePassed = false;
  gameState = GAME_PLAYING;
}

void handleGameInput() {
  // Jump menggunakan pin UP (BTN_UP = 2)
  if (buttonPressed(BTN_UP) && !isJumping && !isDucking) {
    isJumping = true;
    dinoVelocity = JUMP_FORCE;
    // Suara lompat (opsional)
    tone(BUZZER_PIN, 1000, 50);
  }
  
  // Duck menggunakan pin DOWN (BTN_DOWN = 3)
  if (!digitalRead(BTN_DOWN) && !isJumping) {
    isDucking = true;
    dinoHeight = 8; // Mengecil saat menunduk
  } else {
    isDucking = false;
    dinoHeight = 15; // Kembali normal
  }
}

void updateGame() {
  // Update dino physics
  if (isJumping) {
    dinoY += dinoVelocity;
    dinoVelocity += GRAVITY;
    
    // Cek jika sudah menyentuh tanah
    if (dinoY >= GROUND_Y) {
      dinoY = GROUND_Y;
      isJumping = false;
      dinoVelocity = 0;
    }
  }
  
  // Update obstacle position
  obstacleX -= obstacleSpeed;
  
  // Reset obstacle jika sudah melewati layar
  if (obstacleX < -obstacleWidth) {
    obstacleX = random(130, 180); // Muncul lagi dengan jarak random
    obstaclePassed = false;
    
    // Tambah skor setiap obstacle berhasil dilewati
    score += 10;
    
    // Tingkatkan kesulitan seiring waktu
    if (score > 500) {
      obstacleSpeed = 5;
    } else if (score > 1000) {
      obstacleSpeed = 7;
    }
    else if (score > 9999) {
      obstacleSpeed = 9;
    }
  }
  
  // Tambah skor berdasarkan waktu
  if (millis() - lastScoreTime > 100) {
    score++;
    lastScoreTime = millis();
  }
  
  // Collision detection
  int dinoRectX = 20;
  int dinoRectY = dinoY - dinoHeight;
  int dinoRectWidth = dinoWidth;
  int dinoRectHeight = dinoHeight;
  
  int obstacleRectX = obstacleX;
  int obstacleRectY = GROUND_Y - obstacleHeight;
  int obstacleRectWidth = obstacleWidth;
  int obstacleRectHeight = obstacleHeight;
  
  // Simple AABB collision detection
  if (dinoRectX < obstacleRectX + obstacleRectWidth &&
      dinoRectX + dinoRectWidth > obstacleRectX &&
      dinoRectY < obstacleRectY + obstacleRectHeight &&
      dinoRectY + dinoRectHeight > obstacleRectY) {
    gameOver();
  }
}

void gameOver() {
  gameState = GAME_OVER;
  if (score > highScore) {
    highScore = score;
    // Simpan langsung tanpa begin ulang
    preferences.putInt("HIGHSCORE", highScore);
  }
  // Suara game over
  tone(BUZZER_PIN, 200, 200);
  delay(200);
  tone(BUZZER_PIN, 100, 400);
  delay(400);
  noTone(BUZZER_PIN);
  
  // RGB effect saat game over
  for (int i = 0; i < 3; i++) {
    setRGBColor(0); // Merah
    delay(100);
    pixels.clear();
    pixels.show();
    delay(100);
  }
  // Kembali ke mode semula
  updateRGB();
}

/* ===================== EMOJI SYSTEM MANAGEMENT ===================== */
void updateEmojiSystem() {
  unsigned long currentTime = millis();
  
  // Deteksi tombol SELECT
  if (buttonPressed(BTN_SELECT) && !selectPressed) {
    emojiSelectPressend = true;
    lastEmojiSelectTime = currentTime;
    
    if (currentEmojiState == EMOJI_MAIN_MENU) {
      // Dari menu utama, masuk ke sequence dengan tetap menampilkan emopre
      currentEmojiState = EMOJI_IN_SEQUENCE;
      sequenceIndex = 0;
      sequenceForward = true;
      inRandomMode = false;
      lastSequenceTime = currentTime;
      Serial.println("Emoji: Enter sequence mode - showing emopre");
    }
    else if (currentEmojiState == EMOJI_IN_SEQUENCE || currentEmojiState == EMOJI_RANDOM_SEQUENCE) {
      // Jika sudah dalam sequence, tekan SELECT untuk random mode
      currentEmojiState = EMOJI_RANDOM_SEQUENCE;
      inRandomMode = true;
      // Pilih sequence random
      currentSequence = (EmojiSequence)random(0, 3);
      sequenceIndex = 0;
      sequenceForward = true;
      lastSequenceTime = currentTime;
      Serial.print("Emoji: Random sequence selected: ");
      Serial.println(currentSequence);
    }
  }
  
  // Reset selectPressed setelah debounce
  if (emojiSelectPressend && (currentTime - lastEmojiSelectTime > SELECT_DEBOUNCE)) {
    emojiSelectPressend = false;
  }
  
  // Deteksi tombol BACK
  if (buttonPressed(BTN_BACK)) {
    currentEmojiState = EMOJI_MAIN_MENU;
    inRandomMode = false;
    Serial.println("Emoji: Back to main menu");
  }
  
  // Update sequence jika dalam mode sequence
  if ((currentEmojiState == EMOJI_IN_SEQUENCE || currentEmojiState == EMOJI_RANDOM_SEQUENCE) && 
      (currentTime - lastSequenceTime > SEQUENCE_DELAY)) {
    
    // Deteksi tombol UP/DOWN untuk navigasi berurutan
    if (!inRandomMode) {
      if (buttonPressed(BTN_UP)) {
        // Pindah ke sequence berikutnya
        currentSequence = (EmojiSequence)((currentSequence + 1) % 3);
        sequenceIndex = 0;
        sequenceForward = true;
        Serial.print("Emoji: Sequence changed to ");
        Serial.println(currentSequence);
      }
      if (buttonPressed(BTN_DOWN)) {
        // Pindah ke sequence sebelumnya
        currentSequence = (EmojiSequence)((currentSequence + 2) % 3);
        sequenceIndex = 0;
        sequenceForward = true;
        Serial.print("Emoji: Sequence changed to ");
        Serial.println(currentSequence);
      }
    }
    
    // Update sequence index untuk animasi
    lastSequenceTime = currentTime;
    
    // Logika sequence berdasarkan mode
    if (inRandomMode) {
      // Dalam mode random, sequence berjalan otomatis
      if (sequenceForward) {
        sequenceIndex++;
        // Cek apakah sudah sampai akhir sequence
        if (sequenceIndex > 5) { // 0:emonet, 1:yawn1/mad1/cry1, 2:yawn2/mad2/cry2, 3:yawn1/mad1/cry1, 4:emonet, 5:emopre
          sequenceForward = false;
          sequenceIndex = 4; // Kembali ke emonet
        }
      } else {
        sequenceIndex--;
        if (sequenceIndex < 0) {
          sequenceForward = true;
          sequenceIndex = 1;
        }
      }
    } else {
      // Dalam mode berurutan (manual), hanya maju sampai emopre lalu reset
      sequenceIndex++;
      if (sequenceIndex > 5) {
        sequenceIndex = 0;
      }
    }
  }
}

/* ===================== FUNGSI ALARM BARU ===================== */
void checkAlarm() {
  if (!rtc.begin()) return;
  
  DateTime now = rtc.now();
  int currentSecond = now.second();
  
  // Cek apakah alarm aktif dan sudah lewat durasi
  if (alarmActive) {
    unsigned long elapsedTime = (millis() - alarmStartTime) / 1000;
    if (elapsedTime >= alarmDuration) {
      stopAlarm();
      Serial.println("Alarm stopped after duration");
    }
    return;
  }
  
  // Cek setiap detik (hanya ketika detik berubah)
  if (currentSecond != lastCheckedSecond) {
    lastCheckedSecond = currentSecond;
    
    // Cek apakah waktu sekarang sama dengan waktu alarm
    if (now.hour() == alarmHour && now.minute() == alarmMinute && currentSecond == 0) {
      // Alarm waktu tepat
      triggerAlarm();
      Serial.println("Alarm triggered!");
    }
  }
}

void triggerAlarm() {
  alarmActive = true;
  alarmStartTime = millis();
  
  // Bunyikan alarm sesuai dengan suara yang dipilih
  Serial.print("Playing alarm sound: ");
  Serial.println(soundMenuList[alarmSound]);
  
  // Mainkan alarm dalam loop sampai durasi habis
  unsigned long alarmEndTime = millis() + (alarmDuration * 1000);
  
  while (millis() < alarmEndTime && alarmActive) {
    playAlarmSound(alarmSound);
    
    // Cek apakah ada tombol yang ditekan untuk mematikan alarm lebih awal
    if (buttonPressed(BTN_SELECT) || buttonPressed(BTN_BACK)) {
      stopAlarm();
      Serial.println("Alarm stopped by user");
      break;
    }
    
    delay(100); // Jeda kecil
  }
  
  alarmActive = false;
}

void stopAlarm() {
  alarmActive = false;
  noTone(BUZZER_PIN);
}

/* ===================== AUDIO ===================== */
void setupI2S() {
  i2s_config_t config = {
    .mode = (i2s_mode_t)(I2S_MODE_MASTER | I2S_MODE_RX),
    .sample_rate = 16000,
    .bits_per_sample = I2S_BITS_PER_SAMPLE_32BIT,
    .channel_format = I2S_CHANNEL_FMT_ONLY_LEFT,
    .communication_format = I2S_COMM_FORMAT_STAND_I2S,  // Perbaiki: I2S_COMM_FORMAT_STAND_I2S
    .intr_alloc_flags = ESP_INTR_FLAG_LEVEL1,
    .dma_buf_count = 4,
    .dma_buf_len = 256,
    .use_apll = false,
    .tx_desc_auto_clear = false,
    .fixed_mclk = 0
  };

  i2s_pin_config_t pin_config = {
    .bck_io_num = I2S_SCK,
    .ws_io_num = I2S_WS,
    .data_out_num = I2S_PIN_NO_CHANGE,
    .data_in_num = I2S_SD_L  // Gunakan I2S_SD_L sebagai data in
  };

  i2s_driver_install(I2S_NUM_0, &config, 0, NULL);
  i2s_set_pin(I2S_NUM_0, &pin_config);
}

int getSoundLevel() {
  size_t bytesRead;
  esp_err_t err = i2s_read(I2S_NUM_0, audioBuffer, sizeof(audioBuffer), &bytesRead, portMAX_DELAY);  // Perbaiki: portMAX_DELAY
  if (err != ESP_OK) {
    return 0;
  }

  long sum = 0;
  int samples = bytesRead / 4;
  for (int i = 0; i < samples; i++) {
    sum += abs(audioBuffer[i]);
  }
  return samples > 0 ? sum / samples : 0;
}

/* ===================== UTIL ===================== */
bool buttonPressed(uint8_t pin) {
  static uint32_t lastTime[10];
  if (!digitalRead(pin) && millis() - lastTime[pin] > 200) {
    lastTime[pin] = millis();
    return true;
  }
  return false;
}

void handleRTCMenuNavigation() {
  unsigned long currentTime = millis();
  
  // Blink cursor untuk setting
  if (currentTime - lastBlinkTime > 500) {
    lastBlinkTime = currentTime;
    blinkState = !blinkState;
  }
  
  // Handle berdasarkan mode
  if (inTimeSetting) {
    // Setting waktu (jam, menit, detik)
    if (buttonPressed(BTN_UP)) {
      switch(timeSettingPart) {
        case 0: tempHour = (tempHour + 1) % 24; break;
        case 1: tempMinute = (tempMinute + 1) % 60; break;
        case 2: tempSecond = (tempSecond + 1) % 60; break;
      }
    }
    if (buttonPressed(BTN_DOWN)) {
      switch(timeSettingPart) {
        case 0: tempHour = (tempHour + 23) % 24; break;
        case 1: tempMinute = (tempMinute + 59) % 60; break;
        case 2: tempSecond = (tempSecond + 59) % 60; break;
      }
    }
    if (buttonPressed(BTN_SELECT)) {
      timeSettingPart++;
      if (timeSettingPart > 2) {
        saveTimeSetting();
      }
    }
    if (buttonPressed(BTN_BACK)) {
      inTimeSetting = false;
    }
  }
  else if (inCalendarSetting) {
    // Setting tanggal (tahun, bulan, tanggal)
    if (buttonPressed(BTN_UP)) {
      switch(calendarSettingPart) {
        case 0: tempYear++; break;
        case 1: tempMonth = (tempMonth % 12) + 1; break;
        case 2: {
          // Validasi tanggal sesuai bulan
          int maxDay = 31;
          if (tempMonth == 4 || tempMonth == 6 || tempMonth == 9 || tempMonth == 11) maxDay = 30;
          else if (tempMonth == 2) {
            // Cek tahun kabisat
            if ((tempYear % 4 == 0 && tempYear % 100 != 0) || (tempYear % 400 == 0)) maxDay = 29;
            else maxDay = 28;
          }
          tempDay = (tempDay % maxDay) + 1;
          break;
        }
      }
    }
    if (buttonPressed(BTN_DOWN)) {
      switch(calendarSettingPart) {
        case 0: tempYear--; break;
        case 1: tempMonth = (tempMonth == 1) ? 12 : tempMonth - 1; break;
        case 2: {
          int maxDay = 31;
          if (tempMonth == 4 || tempMonth == 6 || tempMonth == 9 || tempMonth == 11) maxDay = 30;
          else if (tempMonth == 2) {
            if ((tempYear % 4 == 0 && tempYear % 100 != 0) || (tempYear % 400 == 0)) maxDay = 29;
            else maxDay = 28;
          }
          tempDay = (tempDay == 1) ? maxDay : tempDay - 1;
          break;
        }
      }
    }
    if (buttonPressed(BTN_SELECT)) {
      calendarSettingPart++;
      if (calendarSettingPart > 2) {
        saveCalendarSetting();
      }
    }
    if (buttonPressed(BTN_BACK)) {
      inCalendarSetting = false;
    }
  }
  else if (inAlarmSetting) {
    // Setting alarm (jam, menit)
    if (buttonPressed(BTN_UP)) {
      switch(alarmSettingPart) {
        case 0: tempAlarmHour = (tempAlarmHour + 1) % 24; break;
        case 1: tempAlarmMinute = (tempAlarmMinute + 1) % 60; break;
      }
    }
    if (buttonPressed(BTN_DOWN)) {
      switch(alarmSettingPart) {
        case 0: tempAlarmHour = (tempAlarmHour + 23) % 24; break;
        case 1: tempAlarmMinute = (tempAlarmMinute + 59) % 60; break;
      }
    }
    if (buttonPressed(BTN_SELECT)) {
      alarmSettingPart++;
      if (alarmSettingPart > 1) {
        saveAlarmSetting();
      }
    }
    if (buttonPressed(BTN_BACK)) {
      inAlarmSetting = false;
    }
  }
  else if (inAlarmDurationSetting) {
    // Setting durasi alarm dengan scroll
    if (buttonPressed(BTN_UP)) {
      // Naikkan durasi 1 detik (maks 300 detik = 5 menit)
      tempAlarmDuration = (tempAlarmDuration < 1000) ? tempAlarmDuration + 10 : 1000;
      Serial.print("Alarm duration: ");
      Serial.println(tempAlarmDuration);
    }
    if (buttonPressed(BTN_DOWN)) {
      // Turunkan durasi 1 detik (min 1 detik)
      tempAlarmDuration = (tempAlarmDuration > 1) ? tempAlarmDuration - 10 : 1;
      Serial.print("Alarm duration: ");
      Serial.println(tempAlarmDuration);
    }
    if (isButtonDoubleClicked(BTN_BACK) || (buttonPressed(BTN_BACK) && buttonPressed(BTN_SELECT))) {
      // Lompat 10 detik dengan double click BACK atau kombinasi BACK+SELECT
      tempAlarmDuration = (tempAlarmDuration + 120 <= 60) ? tempAlarmDuration + 10 : 100;
      Serial.print("Alarm duration +10: ");
      Serial.println(tempAlarmDuration);
    }
    if (buttonPressed(BTN_SELECT)) {
      saveAlarmDurationSetting();
    }
    if (buttonPressed(BTN_BACK) && !isButtonDoubleClicked(BTN_BACK)) {
      inAlarmDurationSetting = false;
    }
  }
  else if (inSoundSetting) {
    // Setting suara alarm dengan scroll
    if (buttonPressed(BTN_UP)) {
      if (soundMenuIndex > 0) {
        soundMenuIndex--;
        // Update scroll offset jika perlu
        if (soundMenuIndex < soundScrollOffset) {
          soundScrollOffset = soundMenuIndex;
        }
      }
      // Preview suara
      playAlarmSound(soundMenuIndex);
    }
    if (buttonPressed(BTN_DOWN)) {
      if (soundMenuIndex < soundMenuCount - 1) {
        soundMenuIndex++;
        // Update scroll offset jika perlu
        if (soundMenuIndex >= soundScrollOffset + VISIBLE_SOUND_ITEMS) {
          soundScrollOffset = soundMenuIndex - VISIBLE_SOUND_ITEMS + 1;
        }
      }
      // Preview suara
      playAlarmSound(soundMenuIndex);
    }
    if (buttonPressed(BTN_SELECT)) {
      alarmSound = soundMenuIndex;
      inSoundSetting = false;
      Serial.print("Alarm sound saved: ");
      Serial.println(soundMenuList[alarmSound]);
    }
    if (buttonPressed(BTN_BACK)) {
      inSoundSetting = false;
    }
  }
  else {
    // Menu utama RTC
    if (buttonPressed(BTN_DOWN)) {
      rtcMenuIndex = (rtcMenuIndex + 1) % rtcMenuCount;
    }
    if (buttonPressed(BTN_UP)) {
      rtcMenuIndex = (rtcMenuIndex + rtcMenuCount - 1) % rtcMenuCount;
    }
    if (buttonPressed(BTN_SELECT)) {
      switch(rtcMenuIndex) {
        case 0: // Hour
          {
            DateTime now = rtc.now();
            tempHour = now.hour();
            tempMinute = now.minute();
            tempSecond = now.second();
            timeSettingPart = 0;
            inTimeSetting = true;
          }
          break;
        case 1: // Calendar
          {
            DateTime now = rtc.now();
            tempYear = now.year();
            tempMonth = now.month();
            tempDay = now.day();
            calendarSettingPart = 0;
            inCalendarSetting = true;
          }
          break;
        case 2: // Alarm
          tempAlarmHour = alarmHour;
          tempAlarmMinute = alarmMinute;
          alarmSettingPart = 0;
          inAlarmSetting = true;
          break;
        case 3: // Alarm Duration
          tempAlarmDuration = alarmDuration;
          inAlarmDurationSetting = true;
          break;
        case 4: // Sound
          soundMenuIndex = alarmSound;
          soundScrollOffset = 0;
          inSoundSetting = true;
          break;
      }
    }
    if (buttonPressed(BTN_BACK)) {
      exitRTCMenu();
    }
  }
}
